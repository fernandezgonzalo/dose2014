class

	SQLITE_MODIFY_STATEMENT
		redefine

		end

feature -- Measurement

	last_row_id: INTEGER_64
			-- Last inserted row
	
feature {NONE} -- Action handlers

	on_before_execute
			-- Called before a statement has been executed.
		do
			Precursor
			last_row_id := 0.to_integer_64
		end

	on_after_executed
			-- Called after a statement has been executed, successfully or not.

			end
		end
